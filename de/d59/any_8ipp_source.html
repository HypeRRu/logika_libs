75template < typename T >
76T Any::Cast() const
77{
78 LOG_WRITE( LOG_DEBUG, LOCALIZED( "Attempt to cast Any value to " )
79 << ToLocString( GetTypeName( typeid( T ) ) ) );
80 if ( Empty() )
81 {
82 LOG_WRITE( LOG_DEBUG, LOCALIZED( "Any value not set" ) );
83 throw std::runtime_error{ "Attempt to cast empty any-value" };
84 }
85 if ( typeid( T ) != holder_->TypeInfo() )
86 {
87 LOG_WRITE( LOG_DEBUG, LOCALIZED( "Cast Any value to " )
88 << ToLocString( GetTypeName( typeid( T ) ) )
89 << LOCALIZED( " failed. Expected type is " )
90 << ToLocString( GetTypeName( holder_->TypeInfo() ) ) );
91 throw std::runtime_error{ "Bad any cast" };
92 }
93 LOG_WRITE( LOG_DEBUG, LOCALIZED( "Casted Any successfully" ) );
94 return static_cast< Holder< T >* >( holder_.get() )->value;
95} // Cast
